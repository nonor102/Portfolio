作品について

本作品は、UnityとC#を用いて開発した縦スクロール型の2Dアクションゲームです。 プレイヤーはキャラクターを操作し、ステージを上方向へと進んでいきます。

SNS上で一時話題となった「風呂キャンセル界隈」をテーマとし、目的 (ゴール) は風呂に入ること、 ゲームオーバーは風呂に入らず就寝する、または風呂に入るやる気をなくすこととしました。

インターンシップの実習として、技術や知識に差のある3人のチームメンバーと共に協力し、1日は設計、4日で作製をし、完成させることができました。 私ともう一人のメンバーで主にバックエンド処理 (スクリプト系) を担当し、もう二人のメンバーでUI/UXの設計やステージのデザインを担当しました。

自分の担当した部分

私は主にゲームのバックエンド処理を担当し、その中でも基本的なキャラクターの動きやスコア (本作においては上った高さ) を管理するスクリプトを作製しました。 キャラクターの壁や床との衝突判定、移動速度やアイテムなどによるキャラクターの動きの変化などを実装し、ゲームの基本的なプレイ感を作り上げることに注力しました。 ゲームクリア、ゲームオーバー時の処理も担当し、ゲームの根幹を支える部分を作製しました。

また、タイトルシーンも担当し、マウス操作ではなくキャラクターを操作しながらストーリーシーン、ゲームシーン、遊び方の説明シーンへ遷移できるようにしました。

C# 
using System.Text.RegularExpressions;
using TMPro;
using Unity.VisualScripting;
using UnityEngine;
using UnityEngine.Pool;
using UnityEngine.SceneManagement;
using Cysharp.Threading.Tasks;

public class Player : MonoBehaviour
{
    [SerializeField] private AudioSource RootSE;

    [SerializeField] private AudioClip jump;
    [SerializeField] private AudioClip run;
    [SerializeField] private Rigidbody2D rb;
    [SerializeField] private float playerSpeed;
    [SerializeField] private float playerJumpPower;
    [SerializeField] private Animator animator;
    private int count;
    [SerializeField] private TextMeshProUGUI scoreText;
    private bool isJumpTriggered = false;
    bool hasPlayedRunSE = false;
    private bool isGrounded;
    public bool isOntoHowToPlay{ get; private set; }
    public bool isOntoStory{ get; private set; }
    public bool isOntoPlay{ get; private set; }
    private float playerHeight = 0f;

    private PlayerAttack playerAttack;

    void Start()
    {
        rb.constraints = RigidbodyConstraints2D.FreezeRotation;
        playerAttack = GetComponent<PlayerAttack>();
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Space) && isGrounded)
        {
            isJumpTriggered = true;
            RootSE.PlayOneShot(jump); // 飛ぶSE
        }

        if (Input.GetKeyDown(KeyCode.Escape))
        {
            GameManager.Instance.PlayerMotivationSetter();
            Destroy(GetComponent<PlayerHP>());
            SceneManager.LoadScene("UI");
        }

        if (Input.GetKeyDown(KeyCode.Return) && isOntoHowToPlay)
        {
            SceneManager.LoadScene("explanation");
        }
        else if (Input.GetKeyDown(KeyCode.Return) && isOntoPlay)
        {
            SceneManager.LoadScene("MainGame");
        }
        else if (Input.GetKeyDown(KeyCode.Return) && isOntoStory)
        {
            SceneManager.LoadScene("Story Scene");
        }

        playerHeight = transform.position.y + 6.5f; // 最初の地点からの高さ

        if (GameManager.Instance.PlayerMotivation <= 0)
        {
            scoreText.text = "ゲームオーバー! \n やる気が0になってしまった! \n"
                + "到達した高さ: " + (int)playerHeight
                + "\n\n Escキーでタイトルへ";
        }
    }

    void FixedUpdate()
    {
        if (GameManager.Instance.PlayerMotivation <= 0)
        {
            animator.SetBool("isGround", true);
            animator.SetBool("isFastRunning", false);
            animator.SetBool("isRunning", false);
            animator.SetBool("isWalking", false);

            rb.constraints = RigidbodyConstraints2D.None;
            rb.constraints = RigidbodyConstraints2D.FreezePositionX;
            rb.constraints = RigidbodyConstraints2D.FreezePositionY;

            while (transform.eulerAngles.z >= 0 && transform.eulerAngles.z < 90)
            {
                transform.Rotate(0f, 0f, 90f);
            }
        }

        int layerMask = 1 << 7;

        // Raycastによる接地判定（3方向）
        Ray2D rayNormal = new Ray2D(transform.position, Vector2.down);
        Ray2D rayNaname = new Ray2D(transform.position, new Vector2(-1, -1));
        Ray2D rayNaname2 = new Ray2D(transform.position, new Vector2(1, -1));
        Debug.DrawRay(rayNormal.origin, rayNormal.direction * 0.3f, Color.red, 0.05f);
        Debug.DrawRay(rayNaname.origin, rayNaname.direction * 0.5f, Color.red, 0.05f);
        Debug.DrawRay(rayNaname2.origin, rayNaname2.direction * 0.5f, Color.red, 0.05f);
        RaycastHit2D hitNormal = Physics2D.Raycast(rayNormal.origin, rayNormal.direction, 0.3f, layerMask);
        RaycastHit2D hitNaname = Physics2D.Raycast(rayNaname.origin, rayNaname.direction, 0.5f, layerMask);
        RaycastHit2D hitNaname2 = Physics2D.Raycast(rayNaname2.origin, rayNaname2.direction, 0.5f, layerMask);
        isGrounded = hitNormal.collider != null || hitNaname.collider != null || hitNaname2.collider != null;

        if (isGrounded)
        {
            animator.SetBool("isGround", true);

            if (isJumpTriggered)
            {
                animator.SetBool("isGround", false);
                animator.SetBool("isFastRunning", false);
                animator.SetBool("isRunning", false);
                animator.SetBool("isWalking", false);
                rb.linearVelocity = new Vector2(rb.linearVelocity.x,
                    playerJumpPower * GameManager.Instance.PlayerVariablesMultiplierByMotivation);
                isJumpTriggered = false;
            }
        }

        if (GameManager.Instance.PlayerMotivation > 0)
        {
            if (Input.GetKey(KeyCode.RightArrow))
            {
                count++;
                if (count >= 10 && isGrounded)
                {
                    RootSE.Play(); // 走るSE
                    count = 0;
                }

                transform.localScale = new Vector3(
                    -Mathf.Abs(transform.localScale.x),
                    transform.localScale.y, transform.localScale.z);

                // やる気の値に応じてアニメーションを切り替え
                if (GameManager.Instance.PlayerMotivation > 50)
                {
                    animator.SetBool("isFastRunning", true);
                    animator.SetBool("isRunning", false);
                    animator.SetBool("isWalking", false);
                }
                else if (GameManager.Instance.PlayerMotivation > 10)
                {
                    animator.SetBool("isFastRunning", false);
                    animator.SetBool("isRunning", true);
                    animator.SetBool("isWalking", false);
                }
                else
                {
                    animator.SetBool("isFastRunning", false);
                    animator.SetBool("isRunning", false);
                    animator.SetBool("isWalking", true);
                }
                rb.linearVelocity = new Vector2(
                    playerSpeed * GameManager.Instance.PlayerVariablesMultiplierByMotivation,
                    rb.linearVelocity.y);
            }
            else if (Input.GetKey(KeyCode.LeftArrow))
            {
                count++;
                if (count >= 15 && isGrounded)
                {
                    RootSE.Play(); // 走るSE
                    count = 0;
                }

                transform.localScale = new Vector3(
                    Mathf.Abs(transform.localScale.x),
                    transform.localScale.y, transform.localScale.z);

                // やる気の値に応じてアニメーションを切り替え
                if (GameManager.Instance.PlayerMotivation > 50)
                {
                    animator.SetBool("isFastRunning", true);
                    animator.SetBool("isRunning", false);
                    animator.SetBool("isWalking", false);
                }
                else if (GameManager.Instance.PlayerMotivation > 10)
                {
                    animator.SetBool("isFastRunning", false);
                    animator.SetBool("isRunning", true);
                    animator.SetBool("isWalking", false);
                }
                else
                {
                    animator.SetBool("isFastRunning", false);
                    animator.SetBool("isRunning", false);
                    animator.SetBool("isWalking", true);
                }
                rb.linearVelocity = new Vector2(
                    -playerSpeed * GameManager.Instance.PlayerVariablesMultiplierByMotivation,
                    rb.linearVelocity.y);
            }
            else
            {
                animator.SetBool("isFastRunning", false);
                animator.SetBool("isRunning", false);
                animator.SetBool("isWalking", false);
            }
        }
    }

    void OnCollisionEnter2D(Collision2D collision)
    {
        if (collision.gameObject.CompareTag("Oyatu"))
        {
            GameManager.Instance.PlayerMotivateValueChanger("Oyatu");
        }
        else if (collision.gameObject.CompareTag("Smartphone"))
        {
            GameManager.Instance.PlayerMotivateValueChanger("Smartphone");
        }
        else if (collision.gameObject.CompareTag("IdleFacePicture"))
        {
            GameManager.Instance.PlayerMotivateValueChanger("IdleFacePicture");
        }
    }

    void OnTriggerEnter2D(Collider2D collision)
    {
        // --- メニュー選択の判定 ---
        if (collision.gameObject.CompareTag("PlayGame"))
        {
            isOntoPlay = true;
            isOntoHowToPlay = false;
            isOntoStory = false;
        }
        else if (collision.gameObject.CompareTag("LoadStory"))
        {
            isOntoPlay = false;
            isOntoHowToPlay = false;
            isOntoStory = true;
        }
        else if (collision.gameObject.CompareTag("HowToPlay"))
        {
            isOntoPlay = false;
            isOntoHowToPlay = true;
            isOntoStory = false;
        }
        // --- アイテム取得の判定 ---
        else if (collision.gameObject.CompareTag("ItemShampoo"))
        {
            if (playerAttack != null)
            {
                playerAttack.SwitchWeapon(PlayerAttack.WeaponType.SoapDispenser);
            }
            Destroy(collision.gameObject);
        }
        else if (collision.gameObject.CompareTag("ItemPerfume"))
        {
            if (playerAttack != null)
            {
                playerAttack.SwitchWeapon(PlayerAttack.WeaponType.MirrorLight);
            }
            Destroy(collision.gameObject);
        }
        // --- その他のギミック判定 ---
        else if (collision.gameObject.CompareTag("Bed"))
        {
            GameManager.Instance.PlayerMotivateValueChanger("Bed");
        }
        else if (collision.gameObject.CompareTag("Death"))
        {
            GameManager.Instance.PlayerMotivateValueChanger("Death");
        }

        if (collision.gameObject.CompareTag("Goal"))
        {
            rb.constraints = RigidbodyConstraints2D.FreezeAll;
            scoreText.text = "ゲームクリア!\n 到達した高さ: " + (int)playerHeight
                + "\n やる気: " + GameManager.Instance.PlayerMotivation
                + "\n\n Escキーでタイトルへ";
        }
    }
}

解説

Player は、プレイヤーキャラクターの移動・ジャンプ・衝突判定・シーン遷移など、 ゲームの中核となる操作ロジックをすべて管理するスクリプトです。 Update でキー入力の検知とゲームオーバー判定を行い、 FixedUpdate で物理演算に基づいた移動処理を実行しています。

「やる気」という独自パラメータ（GameManager.Instance.PlayerMotivation）に応じて 移動速度やアニメーションが3段階で変化する仕組みを実装しています。 また、OnCollisionEnter2D と OnTriggerEnter2D で アイテム取得・メニュー選択・ギミック接触といった多様なイベントを処理しています。

📐 Raycast による接地判定：真下・左斜め下・右斜め下の3方向にRayを飛ばすことで、足場の端でもジャンプ判定が正しく行われるよう工夫しました。
🎭 やる気連動アニメーション：「やる気」の値に応じて「高速走り」「通常走り」「歩き」の3段階でアニメーションを切り替え、ゲーム状態を視覚的に表現しました。
🎮 タイトル画面のメニュー選択：キャラクターを操作して特定エリアに入ることでシーンを遷移する、ゲームらしいメニュー選択方式を実装しました。
⚡ Update と FixedUpdate の使い分け：入力検知は Update で、物理演算を伴う移動処理は FixedUpdate で行い、安定した動作を実現しました。
